Basic Claude API Call
import anthropic
from dotenv import load_dotenv

load_dotenv()
client = anthropic.Anthropic()  # reads ANTHROPIC_API_KEY from env

response = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=1024,
    system="You are a helpful assistant.",  # optional
    messages=[
        {"role": "user", "content": "What is the capital of France?"}
    ]
)

print(response.content[0].text)  # → "The capital of France is Paris."
print(response.usage)             # → input_tokens, output_tokens
Pydantic Model
from pydantic import BaseModel
from typing import Optional, List

# Define your schema
class Invoice(BaseModel):
    vendor: str
    amount: float
    date: str
    line_items: List[str] = []
    notes: Optional[str] = None

# Parse and validate
data = {"vendor": "Acme", "amount": 99.0, "date": "2024-01"}
invoice = Invoice(**data)
print(invoice.vendor)   # → "Acme"
print(invoice.model_dump())  # → dict

Retry Pattern for API Calls
import time
import anthropic

def call_with_retry(client, **kwargs, max_retries: int = 3):
    for attempt in range(max_retries):
        try:
            return client.messages.create(**kwargs)
        except anthropic.RateLimitError:
            if attempt == max_retries - 1: raise
            time.sleep(2 ** attempt)          # exponential backoff: 1s, 2s, 4s
        except anthropic.APIStatusError as e:
            if e.status_code < 500: raise    # client errors — don't retry
            if attempt == max_retries - 1: raise
            time.sleep(2 ** attempt)
RateLimitError → back off and retry. 4xx errors → fix the request, don't retry. 5xx errors → retry with backoff.
